\section{Reproducibility}
\label{app:repro}

\paragraph{Public artifacts.}
\begin{itemize}\setlength\itemsep{1pt}
\item \href{https://github.com/jeffliulab/real-time-power-predict}{\texttt{github.com/jeffliulab/real-time-power-predict}}
   — training code, model checkpoint, deployment app source, all
   experiment scripts.
\item \href{https://github.com/jeffliulab/new-england-real-time-power-predict-data}{\texttt{...new-england-real-time-power-predict-data}}
   — auxiliary data repo; GitHub Actions cron writes the
   daily-refreshed 7-day backtest JSON here.
\item \href{https://huggingface.co/spaces/jeffliulab/predict-power}{\texttt{hf.co/spaces/jeffliulab/predict-power}}
   — public Gradio demo on HF Spaces' free CPU tier; serves real-time
   forecasts and renders the rolling backtest from the data repo.
\item \href{https://huggingface.co/jeffliulab/predict-power-baseline}{\texttt{hf.co/jeffliulab/predict-power-baseline}}
   — trained baseline checkpoint with model card.
\end{itemize}

\paragraph{Build the paper.}
\begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
$ git clone https://github.com/jeffliulab/real-time-power-predict
$ cd real-time-power-predict/report/arxiv
$ latexmk -xelatex -interaction=nonstopmode -halt-on-error paper.tex
\end{lstlisting}

\paragraph{Reproduce the experiments.}
\begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
# Validation experiment (§3): re-run the deployed pipeline on Dec 30, 2022
$ python scripts/validation/reproduce_dec30_2022.py
$ python scripts/validation/augment_validation_with_ci.py

# Multi-window deployment-drift experiment (§4)
$ python scripts/experiments/historical_drift_sweep.py
# (~75 min per window, dominated by HRRR S3 fetches; 8-way parallel)

# Future-weather shift study (App. E)
$ python scripts/validation/future_weather_shift_quantify.py

# Figures
$ python scripts/figures/render_validation_dec30.py
$ python scripts/figures/render_multi_window_overall_mape.py
$ python scripts/figures/render_multi_window_per_zone_mape.py
$ python scripts/figures/render_per_day_mape_timeline.py
$ python scripts/figures/render_btm_solar_correlation.py
\end{lstlisting}

\paragraph{Numeric claims $\to$ source data.}
Every numeric claim in §\ref{sec:validation}--§\ref{sec:btm} traces to a
JSON file in \texttt{report/arxiv/data/}: \texttt{validation\_dec30\_2022.json}
(§3), \texttt{multi\_window\_results.json} (§4), \texttt{btm\_correlation.json}
(§5), \texttt{future\_weather\_shift.json} (Appendix~\ref{app:forecast_shift}).
The JSON files are committed to the repo and are byte-identical
provided HRRR's S3 archive remains stable.
